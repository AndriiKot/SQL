SELECT  RAPlACE(f.title,'Ace Goldfinger', 'ZZZ')
FROM film f
order by f.title;


-- !!!! RAPlACE не поддерживает регулярные выражения !!!

-- на которой было, только будет уже с новым названием 'ZZZ
