-- Итоговые функции
-- Эти функции обычно возвращают результат гораздо быстрее,
-- чем если бы аналогичные вычисления выполнялись в клиентском приложении.


AVG() 
COUNT()
MAX()
MIN()
SUM()

AVG()       функция для вычисления среднего значения(NULL игнорируется)
COUNT()     функция подсчитывает число строк(COUNT() игнорирует значения NULL,!!! (*) Учитываются значения NULL !!!)
MAX()       возвращает наибольшее значения в указанном столбце(в качестве аргумента указывается имя столбца)
MIN()       возвращает наименьшие значения в указанном столбце(в качестве аргумента указывается имя столбца)
SUM()       возвращает сумму значений  в указанном столбце(NULL игнорируется)

SELECT COUNT(*) AS num_items,
       MIN(prod_price) AS price_min,
       MAX(prod_price) AS price_max,
       AVG(prod_price) AS proce_avg
FROM Products;
